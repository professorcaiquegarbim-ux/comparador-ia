async function analisar() { const antes = document.getElementById("antes").files[0]; const depois = document.getElementById("depois").files[0]; if (!antes || !depois) { alert("Envie as duas imagens"); return; } const readerAntes = new FileReader(); const readerDepois = new FileReader(); readerAntes.onload = async function() { readerDepois.onload = async function() { document.getElementById("resultado").innerText = "Analisando..."; const response = await fetch("/api/analisar", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ imagemAntes: readerAntes.result, imagemDepois: readerDepois.result }) }); const data = await response.json(); document.getElementById("resultado").innerText = data.resultado || data.erro; }; readerDepois.readAsDataURL(depois); }; readerAntes.readAsDataURL(antes); }
